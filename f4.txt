pipeline {
    agent any

    stages {
        stage('Create File') {
            steps {
                writeFile file: 'app.py', text: 'print("Hello")'
            }
        }

        stage('Sonar Analysis') {
            steps {
                script {
                    def scannerHome = tool 'sonar'

                    withSonarQubeEnv('sonar') {
                        bat "${scannerHome}\\bin\\sonar-scanner.bat -Dsonar.projectKey=demo -Dsonar.sources=."
                    }
                }
            }
        }
    }
}
