pipeline{
    agent any
    stages{
        stage("copy"){
            steps{
                writeFile file: 'app.py',text:'print("hello")'
                
            }
        }
        stage("sonarAnalysis"){
            steps{
                script{
                    def scannerHome=tool 'sonar'
                    withSonarQubeEnv('sonar'){
                        bat "${scannerHome}\\bin\\sonar-scanner.bat -Dsonar.projectKey=de -Dsonar.sources=."
                    }
                }
            }
        }
    }
}
